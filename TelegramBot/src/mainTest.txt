void loop() {
    // --- PARTE 1: GESTIONE MESSAGGI TELEGRAM ---
    TBMessage msg;
    if (myBot.getNewMessage(msg)) {
        // CASO 1: L'utente ha scritto un messaggio di TESTO normale
        if (msg.messageType == MessageText) {
            Serial.println("\n📩 Testo ricevuto: " + msg.text);
            String risposta = "Ciao" + msg.sender.firstName + " " + msg.sender.lastName + "! \nScegli un colore:";


            // Rispondiamo inviando il menu con la nostra tastiera "appiccicata" sotto
            myBot.sendMessage(msg, risposta.c_str(), miaTastiera);
        }

        // CASO 2: L'utente ha PREMUTO UN BOTTONE (Callback Query)
        else if (msg.messageType == MessageQuery) {
            Serial.print("\n👆 Bottone premuto! Il comando nascosto è: ");
            Serial.println(msg.callbackQueryData); // Leggiamo il codice segreto del bottone

            // Decidiamo cosa fare in base al bottone premuto
            if (msg.callbackQueryData == "cmd_verde") {
                Serial.println("Azione: Accendo il LED di Verde!");
                rgbLedWrite(ledPin, 0, 50, 0); 
            }
            
            else if (msg.callbackQueryData == "cmd_rosso") {
                Serial.println("Azione: Accendo il LED di Rosso!");
                rgbLedWrite(ledPin, 50, 0, 0); 
            }
        }
    }
}
